\subsection*{(1)}
\begin{eqnarray*}
        11x^5 & \equiv & 4   \pmod{13} \\
  2^7 (2^y)^5 & \equiv & 2^2 \pmod{13} \\
       7 + 5y & \equiv & 2   \pmod{12} \\
            y & \equiv & 11  \pmod{12} \\
            x & \equiv & 7   \pmod{13}
\end{eqnarray*}
\begin{verbatim}
>> (11 * 7 ** 5) % 13
4
\end{verbatim}
